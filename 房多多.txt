        // Simple Modal Logic
        const inviteBtn = document.getElementById('inviteBtn');
        const inviteModal = document.getElementById('inviteModal');
        const inviteModalContent = document.getElementById('inviteModalContent');
        const closeModalBtn = document.getElementById('closeModalBtn');

        function openModal() {
            inviteModal.classList.remove('hidden');
            // Small delay to allow display:block to apply before animating opacity/transform
            setTimeout(() => {
                inviteModal.classList.remove('opacity-0');
                inviteModalContent.classList.remove('translate-y-full');
            }, 10);
        }

        function closeModal() {
            inviteModal.classList.add('opacity-0');
            inviteModalContent.classList.add('translate-y-full');
            setTimeout(() => {
                inviteModal.classList.add('hidden');
            }, 300); // match transition duration
        }

        inviteBtn.addEventListener('click', openModal);
        closeModalBtn.addEventListener('click', closeModal);
        inviteModal.addEventListener('click', (e) => {
            if (e.target === inviteModal) {
                closeModal();
            }
        });
    </script>

        let currentCheckbox = null;

        function switchTab(tabId) {
            const tabTodoBtn = document.getElementById('tab-todo');
            const tabDoneBtn = document.getElementById('tab-done');
            const contentTodo = document.getElementById('content-todo');
            const contentDone = document.getElementById('content-done');

            // Reset styles
            tabTodoBtn.classList.remove('text-primary');
            tabTodoBtn.classList.add('text-on-surface-variant');
            tabTodoBtn.querySelector('div').classList.add('hidden');
            
            tabDoneBtn.classList.remove('text-primary');
            tabDoneBtn.classList.add('text-on-surface-variant');
            tabDoneBtn.querySelector('div').classList.add('hidden');

            contentTodo.classList.add('hidden');
            contentDone.classList.add('hidden');

            // Apply active styles
            if (tabId === 'todo') {
                tabTodoBtn.classList.remove('text-on-surface-variant');
                tabTodoBtn.classList.add('text-primary');
                tabTodoBtn.querySelector('div').classList.remove('hidden');
                contentTodo.classList.remove('hidden');
            } else {
                tabDoneBtn.classList.remove('text-on-surface-variant');
                tabDoneBtn.classList.add('text-primary');
                tabDoneBtn.querySelector('div').classList.remove('hidden');
                contentDone.classList.remove('hidden');
            }
        }

        function openModal(checkbox) {
            if (checkbox.checked) {
                currentCheckbox = checkbox;
                const modal = document.getElementById('completion-modal');
                const card = document.getElementById('modal-card');
                
                modal.classList.remove('hidden');
                // Trigger reflow for transition
                void modal.offsetWidth;
                
                modal.classList.remove('opacity-0');
                card.classList.remove('scale-95');
                card.classList.add('scale-100');
            }
        }

        function closeModal() {
            const modal = document.getElementById('completion-modal');
            const card = document.getElementById('modal-card');
            
            modal.classList.add('opacity-0');
            card.classList.remove('scale-100');
            card.classList.add('scale-95');
            
            setTimeout(() => {
                modal.classList.add('hidden');
                if (currentCheckbox) {
                    currentCheckbox.checked = false; // Reset checkbox if cancelled
                    currentCheckbox = null;
                }
            }, 300);
        }
